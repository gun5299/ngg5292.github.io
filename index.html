function getQueryParams() { const params = new URLSearchParams(window.location.search); return { date: params.get('date'), // 디데이 날짜 text: params.get('text'), // 텍스트 색상 background: params.get('background'), // 배경색상 }; } function calculateDday(targetDate) { const now = new Date().getTime(); const goal = new Date(targetDate).getTime(); const distance = now - goal; const days = Math.floor(distance / (1000 * 60 * 60 * 24)); return days >= 0 ? `D+${days}` : `D-${Math.abs(days)}`; } function applyCustomizations(params) { // URL 매개변수에서 스타일 정보 적용 if (params.background) { document.body.style.backgroundColor = `#${params.background}`; } if (params.text) { document.body.style.color = `#${params.text}`; } // 디데이 계산 및 표시 const countElement = document.getElementById('count'); if (params.date) { countElement.textContent = calculateDday(params.date); } else { countElement.textContent = "날짜를 설정하세요!"; } } // main script const params = getQueryParams(); applyCustomizations(params); // 실시간 업데이트 (1초 간격) setInterval(() => { if (params.date) { document.getElementById('count').textContent = calculateDday(params.date); } }, 1000);
